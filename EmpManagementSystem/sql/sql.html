CREATE DATABASE empmanagement;
        USE empmanagement;

        -- Create the 'users' table
        CREATE TABLE users (
        id INT IDENTITY(1,1) PRIMARY KEY,
        role_type NVARCHAR(5) NOT NULL CHECK (role_type IN ('emp', 'admin')),
        username NVARCHAR(100) NOT NULL,
        email NVARCHAR(150) UNIQUE NOT NULL,
        phone NVARCHAR(15) NOT NULL,
        uaddress NVARCHAR(MAX) NULL,
        pin NVARCHAR(10) NULL,
        passwd NVARCHAR(255) NOT NULL,
        parent_id INT NULL,
        insert_date DATETIME DEFAULT GETDATE(),
        update_date DATETIME DEFAULT GETDATE()
        );

        -- Create the 'projects' table
        CREATE TABLE projects (
        id INT IDENTITY(1,1) PRIMARY KEY,
        project_name NVARCHAR(150) NOT NULL,
        project_details NVARCHAR(MAX),
        insert_date DATETIME DEFAULT GETDATE(),
        update_date DATETIME DEFAULT GETDATE()
        );

        -- Create the 'assign_project' table
        CREATE TABLE assign_project (
        id INT IDENTITY(1,1) PRIMARY KEY,
        user_id INT NOT NULL,
        project_id INT NOT NULL,
        assigned_by INT NOT NULL,
        assign_date DATETIME DEFAULT GETDATE()
        );
